 30:  * A StreamedResponse uses a callback for its content.

 46:      * Constructor.

 51:      */

 53:         parent::__construct(null, $status, $headers);
 54:

 57:         }
 58:         $this->streamed = false;
 59:     }

 66:     }

 76:         if (!is_callable($callback)) {
 77:             throw new LogicException('The Response callback must be a valid PHP callable.');
