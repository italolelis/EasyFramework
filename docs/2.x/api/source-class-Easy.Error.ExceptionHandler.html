 61:      *
 62:      * @return Response A Response instance
 63:      */
 64:     public function createResponse($exception)
 65:     {
 66:         $content = '';
 67:         $title = '';
 68:         try {
 69:             if (!$exception instanceof FlattenException) {
 70:                 $exception = FlattenException::create($exception);
 71:             }
 72:             
 73:             switch ($exception->getStatusCode()) {
 74:                 case 404:
 75:                     $title = 'Sorry, the page you are looking for could not be found.';
 76:                     break;
 77:                 default:
 78:                     $title = 'Whoops, looks like something went wrong.';
 79:             }
 80: 
 81:             if ($this->debug) {
 82:                 $content = $this->getContent($exception);
 83:             }
 84:         } catch (Exception $e) {
 85:             // something nasty happened and we cannot throw an exception here anymore
 86:             if ($this->debug) {
 87:                 $title = sprintf('Exception thrown when handling an exception (%s: %s)', get_class($exception), $exception->getMessage());
 88:             } else {
 89:                 $title = 'Whoops, looks like something went wrong.';
 90:             }
 91:         }
 92: 
 93:         return new Response($this->decorate($content, $title), $exception->getStatusCode(), $exception->getHeaders());
 94:     }

221:     /**
222:      * Formats an array as a string.
223:      *
224:      * @param array $args The argument array
225:      *
226:      * @return string
227:      */
228:     public function formatArgs(array $args)
229:     {
230:         $result = array();
231:         foreach ($args as $key => $item) {
232:             if ('object' === $item[0]) {
233:                 $formattedValue = sprintf("<em>object</em>(%s)", $this->abbrClass($item[1]));
234:             } elseif ('array' === $item[0]) {
235:                 $formattedValue = sprintf("<em>array</em>(%s)", is_array($item[1]) ? $this->formatArgs($item[1]) : $item[1]);
236:             } elseif ('string' === $item[0]) {
237:                 $formattedValue = sprintf("'%s'", htmlspecialchars($item[1], ENT_QUOTES | ENT_SUBSTITUTE, $this->charset));
238:             } elseif ('null' === $item[0]) {
239:                 $formattedValue = '<em>null</em>';
240:             } elseif ('boolean' === $item[0]) {
241:                 $formattedValue = '<em>' . strtolower(var_export($item[1], true)) . '</em>';
242:             } elseif ('resource' === $item[0]) {
243:                 $formattedValue = '<em>resource</em>';
244:             } else {
245:                 $formattedValue = str_replace("\n", '', var_export(htmlspecialchars((string) $item[1], ENT_QUOTES | ENT_SUBSTITUTE, $this->charset), true));
246:             }
247: 
248:             $result[] = is_int($key) ? $formattedValue : sprintf("'%s' => %s", $key, $formattedValue);
249:         }
250: 
251:         return implode(', ', $result);
252:     }
