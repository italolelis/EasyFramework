24:  * UrlGeneratorInterface is the interface that all URL generator classes must implement.
25:  *
26:  * The constants in this interface define the different types of resource references that
27:  * are declared in RFC 3986: http://tools.ietf.org/html/rfc3986
28:  * We are using the term "URL" instead of "URI" as this is more common in web applications
29:  * and we do not need to distinguish them as the difference is mostly semantical and
30:  * less technical. Generating URIs, i.e. representation-independent resource identifiers,
31:  * is also possible.
32:  *
33:  * @author √çtalo Lelis de Vietro <italolelis@gmail.com>
34:  */
35: interface IUrlGenerator
36: {
37:     /**
38:      * Generates an absolute URL, e.g. "http://example.com/dir/file".
39:      */
40: 
41:     const ABSOLUTE_URL = true;
42: 
43:     /**
44:      * Generates an absolute path, e.g. "/dir/file".
45:      */
46:     const ABSOLUTE_PATH = false;
47: 
48:     /**
49:      * Generates a relative path based on the current request path, e.g. "../parent-file".
50:      * @see UrlGenerator::getRelativePath()
51:      */
52:     const RELATIVE_PATH = 'relative';
53: 
54:     /**
55:      * Generates a network path, e.g. "//example.com/dir/file".
56:      * Such reference reuses the current scheme but specifies the host.
57:      */
58:     const NETWORK_PATH = 'network';
59:

64:      * $param mixed $params The params to the action
65:      * @return string An absolute url to the action
66:      */
67:     public function create($actionName, $controllerName = null, $params = null, $area = true, $full = true);
