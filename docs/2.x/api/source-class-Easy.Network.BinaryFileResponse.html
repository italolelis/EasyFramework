161:     public function prepare(Request $request) {
162:         $this->headers->set('Content-Length', $this->file->getSize());
163:         $this->headers->set('Accept-Ranges', 'bytes');
164:         $this->headers->set('Content-Transfer-Encoding', 'binary');
165: 
166:         if (!$this->headers->has('Content-Type')) {
167:             $this->headers->set('Content-Type', $this->file->getMimeType() ? : 'application/octet-stream');
168:         }
169: 
170:         if ('HTTP/1.0' != $request->server->get('SERVER_PROTOCOL')) {
171:             $this->setProtocolVersion('1.1');
172:         }
173: 
174:         $this->offset = 0;
175:         $this->maxlen = -1;
176: 
177:         if (self::$trustXSendfileTypeHeader && $request->headers->has('X-Sendfile-Type')) {
178:             // Use X-Sendfile, do not send any content.
179:             $type = $request->headers->get('X-Sendfile-Type');
180:             $path = $this->file->getRealPath();
181:             if (strtolower($type) == 'x-accel-redirect') {
182:                 // Do X-Accel-Mapping substitutions.
183:                 foreach (explode(',', $request->headers->get('X-Accel-Mapping', '')) as $mapping) {
184:                     $mapping = explode('=', $mapping, 2);
185: 
186:                     if (2 == count($mapping)) {
187:                         $location = trim($mapping[0]);
188:                         $pathPrefix = trim($mapping[1]);
189: 
190:                         if (substr($path, 0, strlen($pathPrefix)) == $pathPrefix) {
191:                             $path = $location . substr($path, strlen($pathPrefix));
192:                             break;
193:                         }
194:                     }
195:                 }
196:             }
197:             $this->headers->set($type, $path);
198:             $this->maxlen = 0;
199:         } elseif ($request->headers->has('Range')) {
200:             // Process the range headers.
201:             if (!$request->headers->has('If-Range') || $this->getEtag() == $request->headers->get('If-Range')) {
202:                 $range = $request->headers->get('Range');
203: 
204:                 list($start, $end) = array_map('intval', explode('-', substr($range, 6), 2)) + array(0);
205: 
206:                 if ('' !== $end) {
207:                     $this->maxlen = $end - $start;
208:                 } else {
209:                     $end = $this->file->getSize() - 1;
210:                 }
211: 
212:                 $this->offset = $start;
213: 
214:                 $this->setStatusCode(206);
215:                 $this->headers->set('Content-Range', sprintf('bytes %s-%s/%s', $start, $end, $this->file->getSize()));
